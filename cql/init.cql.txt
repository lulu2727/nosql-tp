CREATE KEYSPACE IF NOT EXISTS air_data WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };

USE air_data;

CREATE TABLE air_data.measures (
  date timestamp,
  station text,
  no int,
  no2 int,
  pm10 int,
  pm25 int,
  co2 int,
  temp int,
  humi int,
  PRIMARY KEY (station, date)
);
